personas = [{'nombre':'Alicia','edad':30},{'nombre':'Roberto','edad':25},
    {'nombre':'Carolina','edad':35}]

def getEdad(persona):
    return persona['edad']

personaOrdenada = sorted(personas,key=getEdad)
print(personaOrdenada)
