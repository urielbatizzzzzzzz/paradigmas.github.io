class CompraCatalogo:
    def inicio(lista):
        lista.items = []
    def agregarElem(lista,nombreElem,precio):
        lista.items.append({"nombre":nombreElem,"precio":precio})
        print(f""{nombreElem}" fue agregado a su lista de compras.")
    def totalCompras(lista):
        total = sum(elemento[precio] for elemento in lista.items)
        return total
